const auth = getAuth(app); const db = getFirestore(app); // DOM 요소 가져오기 const authSection = document.getElementById('auth-section'); const appSection = document.getElementById('app-section'); const emailInput = document.getElementById('email-input'); const passwordInput = document.getElementById('password-input'); const loginButton = document.getElementById('login-button'); const signupButton = document.getElementById('signup-button'); const logoutButton = document.getElementById('logout-button'); const userEmailSpan = document.getElementById('user-email'); const errorMessage = document.getElementById('error-message'); const titleInput = document.getElementById('event-title'); const dateInput = document.getElementById('event-date'); const addButton = document.getElementById('add-button'); const ddayList = document.getElementById('dday-list'); let currentDdays = []; // 실시간 데이터를 담을 배열 let unsubscribeFromDdays = null; // 실시간 리스너 구독 해제 함수 // --- 인증 관련 함수 --- signupButton.addEventListener('click', async () => { try { const email = emailInput.value; const password = passwordInput.value; await createUserWithEmailAndPassword(auth, email, password); errorMessage.textContent = ''; } catch (error) { errorMessage.textContent = "회원가입 실패: " + error.message; } }); loginButton.addEventListener('click', async () => { try { const email = emailInput.value; const password = passwordInput.value; await signInWithEmailAndPassword(auth, email, password); errorMessage.textContent = ''; } catch (error) { errorMessage.textContent = "로그인 실패: 이메일 또는 비밀번호를 확인하세요."; } }); logoutButton.addEventListener('click', () => { signOut(auth); }); // --- 핵심 로직: 사용자의 로그인 상태 변화 감지 --- onAuthStateChanged(auth, (user) => { if (user) { // 사용자가 로그인한 경우 authSection.classList.add('hidden'); appSection.classList.remove('hidden'); userEmailSpan.textContent = `환영합니다, ${user.email}`; // 해당 사용자의 D-Day 목록만 실시간으로 가져오기 listenToDdays(user.uid); } else { // 사용자가 로그아웃한 경우 authSection.classList.remove('hidden'); appSection.classList.add('hidden'); // 이전에 구독했던 실시간 리스너가 있다면 해제 if (unsubscribeFromDdays) { unsubscribeFromDdays(); } } }); // --- Firestore 데이터베이스 관련 함수 (CRUD) --- function listenToDdays(userId) { const ddaysCollection = collection(db, "ddays"); // 쿼리: 'ddays' 컬렉션에서 'userId'가 현재 로그인한 사용자의 id와 같은 문서들만 선택 const q = query(ddaysCollection, where("userId", "==", userId)); // 실시간 리스너 구독 unsubscribeFromDdays = onSnapshot(q, (snapshot) => { currentDdays = snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() })); renderDdays(); }); } addButton.addEventListener('click', async () => { const title = titleInput.value; const date = dateInput.value; const user = auth.currentUser; if (title && date && user) { try { // 'ddays' 컬렉션에 새 문서 추가 await addDoc(collection(db, "ddays"), { title: title, date: date, userId: user.uid, // ★★★ 문서를 생성한 사용자 ID 저장 createdAt: new Date() }); titleInput.value = ''; dateInput.value = ''; } catch (error) { console.error("Error adding document: ", error); } } }); async function deleteDday(docId) { try { await deleteDoc(doc(db, "ddays", docId)); } catch (error) { console.error("Error removing document: ", error); } } function renderDdays() { ddayList.innerHTML = ''; // 날짜순으로 정렬 const sortedDdays = [...currentDdays].sort((a,b) => new Date(a.date) - new Date(b.date)); sortedDdays.forEach(dday => { const li = document.createElement('li'); li.classList.add('dday-item'); const targetDate = new Date(dday.date); const diff = targetDate - new Date(); const diffDays = Math.ceil(diff / (1000 * 60 * 60 * 24)); let resultText = diff < 0 ? `D+${Math.abs(diffDays-1)}` : `D-${diffDays}`; if (diffDays === 0) resultText = 'D-Day!'; li.innerHTML = `
    ${dday.title} (${dday.date}) ${resultText}
    `; li.querySelector('.delete-button').addEventListener('click', () => deleteDday(dday.id)); ddayList.appendChild(li); }); }
